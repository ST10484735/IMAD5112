package vcmsa.ci.assigment1

import android.annotation.SuppressLint
import android.os.Bundle
import android.view.View
import android.widget.Button
import android.widget.EditText
import android.widget.TextView
import androidx.activity.enableEdgeToEdge
import androidx.appcompat.app.AppCompatActivity
import androidx.core.view.ViewCompat
import androidx.core.view.WindowInsetsCompat

class MainActivity : AppCompatActivity() {


     var enter: EditText? = null
    var food: Button? = null
    var reset: Button? = null
    var output_: TextView? = null

    @SuppressLint("MissingInflatedId")
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        enableEdgeToEdge()
        setContentView(R.layout.activity_main)

        enter = findViewById<View>(R.id.enter) as EditText
        food = findViewById<View>(R.id.food) as Button
        reset = findViewById<View>(R.id.reset) as Button
        output_ = findViewById<View>(R.id.output) as TextView


        enter!!.setOnClickListener{
            val time = enter!!.text.toString()
            val meal:String

            if(time.lowercase() == "morning"){
                meal = "cereal\nbread and tea"
            }

            else if(time.lowercase() == "mid morning"){
                meal = "jungle otes\ntoast"
            }

            else if(time.lowercase() == "afternoon"){
                meal = "rice\npap and steak"
            }

            else if(time.lowercase() == "mid afternoon"){
            meal = "cake\nsandwich and yogurt"
        }
            else if(time.lowercase() == "dinner"){
                meal = "noodles\nmeatballs"
            }

            else{
                meal = "Invalid Input"
            }

            output_!!.text = meal




        }

        reset!!.setOnClickListener(){

            enter!!.setText("")
            output_!!.setText("")
        }



        ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main)) { v, insets ->
            val systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars())
            v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom)
            insets
        }
    }
}
